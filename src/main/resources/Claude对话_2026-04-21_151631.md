```sql
-- 用户表

  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(50) NOT NULL COMMENT '科目名',
  description VARCHAR(200) COMMENT '描述',
  icon VARCHAR(200) COMMENT '图标url'

  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  subject_id BIGINT NOT NULL COMMENT '科目id',

  type TINYINT DEFAULT 0 COMMENT '0单选 1多选 2判断',
```

```java
    private Integer code;

    private T data;

    public static <T> Result<T> success(T data) {
```

JwtUtil.java
java
```java
public class JwtUtil {
    private static final String SECRET = "quizgo-secret-key-2024";
    private static final long EXPIRE = 86400000L;

    public static String generateToken(Long userId, Integer role) {
        return Jwts.builder()
            .claim("userId", userId)
            .claim("role", role)
            .setExpiration(new Date(System.currentTimeMillis() + EXPIRE))
            .signWith(SignatureAlgorithm.HS256, SECRET)
            .compact();
    }

    public static Claims parseToken(String token) {
        return Jwts.parser().setSigningKey(SECRET).parseClaimsJws(token).getBody();
    }

    public static Long getUserId(String token) {
        return Long.valueOf(parseToken(token).get("userId").toString());
    }

    public static Integer getRole(String token) {
        return Integer.valueOf(parseToken(token).get("role").toString());
    }
}
```

JwtInterceptor.java（需要登录的接口都会被拦截）
java
```java
@Component
public class JwtInterceptor implements HandlerInterceptor {
    @Override
    public boolean preHandle(HttpServletRequest req, HttpServletResponse res, Object handler) throws Exception {
        String auth = req.getHeader("Authorization");
        if (auth == null || !auth.startsWith("Bearer ")) {
            res.setStatus(401);
            res.getWriter().write("{\"code\":401,\"msg\":\"未登录\"}");
            return false;
        }
        try {
            Claims claims = JwtUtil.parseToken(auth.substring(7));
            req.setAttribute("userId", Long.valueOf(claims.get("userId").toString()));
            req.setAttribute("role", Integer.valueOf(claims.get("role").toString()));
            return true;
        } catch (Exception e) {
            res.setStatus(401);
            res.getWriter().write("{\"code\":401,\"msg\":\"token无效\"}");
            return false;
        }
    }
}
```

utils/request.js（全局封装，只需写一次）：
javascript
```javascript
const BASE_URL = 'http://localhost:8080'
export function request(options) {
  return new Promise((resolve, reject) => {
    uni.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data || {},
      header: {
        'Authorization': 'Bearer ' + uni.getStorageSync('token'),
        'Content-Type': 'application/json'
      },
      success: res => {
        if (res.data.code === 200) resolve(res.data.data)
        else {
          uni.showToast({ title: res.data.msg || '请求失败', icon: 'none' })
          reject(res.data)
        }
      },
      fail: reject
    })
  })
}
```
